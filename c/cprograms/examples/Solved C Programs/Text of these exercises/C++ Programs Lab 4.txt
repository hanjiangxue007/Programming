		{

		printf("\n");

	//code to add matrices

	}
	printf("The second matrix  is ");
	printf("\n");
	for(r=0;r<3;r++)
	{
		for(c=0;c<3;c++)
		{
			printf("%d\t",m2[r][c]);
		}
		printf("\n");
	}
	//code to multply matrices
	for(r=0;r<3;r++)
	{
		for(c=0;c<3;c++)

			printf("%d\t",m3[r][c]);

		printf("\n");
