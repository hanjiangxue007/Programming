/*3.	Write a C program to swap two values by using function.*/
#include<stdio.h>
#include<conio.h>
void swap(int *,int *);
main()
{
	int x,y;
	printf("Enter the first number : ");
	scanf("%d",&x);
	printf("Enter the second number : ");
	scanf("%d",&y);
	printf("\nThe value of x before swapping =%d ",x);
	printf("\nThe value of y before swapping=%d",y);
	//code to call function
	swap(&x,&y);
	printf("\nThe value of x after swapping =%d ",x);
	printf("\nThe value of y after swapping=%d",y);
	getch();
}
void swap(int *a,int *b)
{
	int t;
	t=*a;
	*a=*b;
	*b=t;
}

/*4.	Write a C program to pass a value to a function and check it is even or odd by using pointer parameter.*/

#include<conio.h>
char *EvenOdd(int );
main()
{
	int n;
	char *a;
	printf("Enter a number ");
	scanf("%d",&n);
	a=EvenOdd(n);
	printf("\n%s",a);
	getch();
}
char *EvenOdd(int n)
{
	int r;
	char *result;
	r=n%2;
	if(r==0)
	{
		result="It is even number";
	}
	else
	{
		result="It is odd number";
	}
	return(result);
}
