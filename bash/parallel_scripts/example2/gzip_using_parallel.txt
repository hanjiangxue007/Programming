The following are equivalent:

  (echo file1; echo file2) | parallel gzip
  parallel gzip ::: file1 file2
  parallel gzip {} ::: file1 file2
  parallel --arg-sep ,, gzip {} ,, file1 file2
  parallel --arg-sep ,, gzip ,, file1 file2
  parallel ::: "gzip file1" "gzip file2"


best way: 
parallel gzip ::: file1 file2


parallel ::: "gzip file1" "gzip file2"
parallel ::: "gunzip file1" "gunzip file2"
